cmake_minimum_required(VERSION 3.126)
project(OpenGL-learn)
cmake_policy(SET CMP0072 NEW)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_MODULE_PATH ${CMAKE_MODULE_PATH} ${CMAKE_CURRENT_SOURCE_DIR}/cmake)


file(GLOB SHADER_FILES
    ${CMAKE_SOURCE_DIR}/shaders/*.glsl
)

foreach(shader_file ${SHADER_FILES})
    add_custom_command(TARGET OpenGL-learn POST_BUILD
        COMMAND ${CMAKE_COMMAND} -E copy_if_different
            ${shader_file}
            $<TARGET_FILE_DIR:OpenGL-learn>/shaders/
    )
endforeach()

find_package(OpenGL REQUIRED)
find_package(glfw3 REQUIRED)

set(SOURCE_FILE src/2-shaders/2.4-uniformOffset.cpp)

add_executable(OpenGL-learn
    ${SOURCE_FILE}
    src/glad.c
)

target_include_directories(OpenGL-learn PRIVATE include lib shaders)


target_link_libraries(OpenGL-learn ${GLFW3_LIBRARY} OpenGL::GL)
